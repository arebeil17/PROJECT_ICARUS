main:
		addi    $t1,    $0,     1		# t1 = 1
		addi    $t2,    $0,     2		# t2 = 2
		addi    $t3,    $0,     3		# t3 = 3
		addi    $t4,    $0,     4		# t4 = 4
		addi    $t5,    $0,     5		# t5 = 5
		addi    $t6,    $0,     6		# t6 = 6
		addi    $t7,    $0,     7       # t7 = 7
		sw    	$0,      0($0)  		# MEM[0] = 0
		sw    	$t1,     4($0)  		# MEM[1] = 1
		sw    	$t2,     8($0)  		# MEM[2] = 2
		sw    	$t3,    12($0)  		# MEM[3] = 3
		sw    	$t4,    16($0)  		# MEM[4] = 4
		sw    	$t5,    20($0)  		# MEM[5] = 5
		sw    	$t6,    24($0)  		# MEM[6] = t6
loop:	
		lw    	$s0,     0($0)  		# s0 = MEM[0]
		lw    	$s1,     4($0)  		# s1 = MEM[1]
		lw    	$s2,     8($0)  		# s2 = MEM[2]
		lw    	$s3,    12($0)  		# s3 = MEM[3]
		add		$s4,	$s3,	$s1		# s4 = s3 + s1
		add		$s5,	$s3,	$s2		# s5 = s3 + s2
		mul     $s6,    $s3,    $s2     # s6 = s3 * s2
		sw    	$s6,    24($0)  		# MEM[6] = s6
		lw    	$s6,    24($0)  		# s6 = MEM[6]
		add		$s7,	$s6,	$s1		# s7 = s6 + s1
		add		$t8,	$s2,	$6		# t8 = s2 + s6		
    	j       loop
